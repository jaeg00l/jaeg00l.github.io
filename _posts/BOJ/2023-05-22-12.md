## 답안

```cpp
#include <iostream>
#include <stdio.h>
#include <queue>
using namespace std;
#define MAX 101

struct Point {
   int row;    // 행
   int col;    // 열
   int dist;   // 거리
};
// 상 하 좌 우 이동할 때, Delta Array
int D[4][2] = { {-1, 0}, {1, 0}, {0, -1}, {0, 1} };
int N, M;
int Board[MAX][MAX];

int bfs(int srcRow, int srcCol, int dstRow, int dstCol) {
   bool visited[MAX][MAX] = { false };
   // 큐를 Point 구조체형으로 만들어서 행,열,거리 정보를 다룰 수 있다
   queue<Point> q;
   visited[srcRow][srcCol] = true;
   q.push( { srcRow, srcCol, 1 });
   while( !q.empty()) {
      Point curr = q.front();
      q.pop();
      if(curr.row == dstRow && curr.col == dstCol)
         return curr.dist;

      for(int i=0; i<4; ++i) {
         int nr = curr.row + D[i][0];
         int nc = curr.col + D[i][1];
         // N x M 행렬을 벗어나는지 확인
         if(nr<1 || nr>N || nc<1 || nc>M) continue;
         // 방문한적 있는지 확인
         if(visited[nr][nc]) continue;
         // 벽인지 확인
         if(Board[nr][nc] == 0) continue;
         // 위 조건 모두 해당되지 않으면 갈 수 있는 길이므로 방문
         visited[nr][nc] = true;
         // push할 때 nr, nc가 위치, 거리 1증가
         q.push( { nr, nc, curr.dist + 1 });
      }
   }
   return -1;
}

int main(void) {
   cin >> N >> M;
   for(int i=1; i<=N; ++i) 
      for(int j=1; j<=M; ++j)
         scanf("%1d", &Board[i][j]);
   
   int srcRow, srcCol, dstRow, dstCol;
   srcRow = 1;
   srcCol = 1;
   dstRow = N;
   dstCol = M;

   cout << bfs(srcRow, srcCol, dstRow, dstCol);
   return 0;
}
```

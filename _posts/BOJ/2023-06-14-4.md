```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int main(void) {
   int N;
   cin >> N; 
   int * arr = new int[N];
   for(int i=0; i<N; ++i) {
      int num;
      cin >> num;
      arr[i] = num;
   }
   sort(arr, arr+N);
   for(int i=0; i<N; ++i)
      cout << arr[i] << endl;
	 delete [] arr;
   return 0;
}
```

## 답안

```cpp
#include <iostream>
using namespace std;

int A, B;
int ans = 0;
int temp;
int main(void) {
   cin >> A >> B;
   int * arr = new int[A];
   for(int i=0; i<A; ++i) {
      int k;
      cin >> k;
      arr[i] = k;
   }
   
   for(int i=0; i<A; ++i) {
      for(int j=i+1; j<A; ++j) {
         for(int k=j+1; k<A; ++k) {
            temp = arr[i]+arr[j]+arr[k];
            if(ans<temp && temp<=B) 
               ans = temp;
         }
      }
      
   }
   
   cout << ans;
   delete [] arr;
   return 0;
}
```

브루트 포스 알고리즘을 이용한 것으로 모든 경우의 수를 다 구한 후 정답을 찾는다.
